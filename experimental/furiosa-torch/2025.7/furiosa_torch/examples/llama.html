import time

import torch
from furiosa import native_torch
from furiosa.native_torch import compiler
from furiosa.torch import cache
from furiosa.torch.extras.transformers import RngdStaticCache, compile_model
from transformers import (
    AutoModelForCausalLM,
    AutoTokenizer,
    GenerationConfig,
    TextStreamer,
)


MODEL = "meta-llama/Llama-3.1-8B-Instruct"
CACHE_KEY = "llama-3.1"
IS_INSTRUCT = True
IS_REMOTE_CODE = False
MAX_SEQ_LEN = 1024
PREFILL_BUCKET_SEQ_LEN = [32, 64, 128, 256, 512, 768]


def choose_bucket(length):
    for bucket in PREFILL_BUCKET_SEQ_LEN:
        if bucket >= length:
            return bucket
    raise Exception()


def warm_up(model, num_batches, prefill_length, seq_length, device):
    past_key_values = RngdStaticCache(
        config=model.config,
        max_batch_size=num_batches,
        max_cache_len=prefill_length,
        device=device,
        dtype=model.dtype,
    )

    # Prefill
    print(f"Preparing Prefill[b{num_batches} {prefill_length}]")
    model(
        input_ids=torch.empty(
            [num_batches, prefill_length], dtype=torch.int32, device=device
        ),
        attention_mask=torch.empty(
            [num_batches, 1, prefill_length, prefill_length],
            dtype=torch.bfloat16,
            device=device,
        ),
        use_cache=True,
        cache_position=torch.zeros([prefill_length], dtype=torch.int32).to(device),
        past_key_values=past_key_values,
        return_dict=True,
    )

    # Decode
    print(f"Preparing Decode[b{num_batches} {seq_length}]")
    past_key_values.resize(seq_length)
    model(
        input_ids=torch.empty([num_batches, 1], dtype=torch.int32, device=device),
        attention_mask=torch.empty(
            [num_batches, 1, 1, seq_length], dtype=torch.bfloat16, device=device
        ),
        use_cache=True,
        cache_position=torch.tensor([0], dtype=torch.int32).to(device),
        past_key_values=past_key_values,
        return_dict=True,
    )


def sample_argmax(logits):
    return logits.cpu().argmax(-1).to(torch.int32).unsqueeze(1)


def sample_multinomial(logits):
    probs = torch.nn.functional.softmax(logits.to(torch.float32), dim=-1)
    return torch.multinomial(probs.cpu(), num_samples=1).to(torch.int32)


def sample_topk(logits, k, min_prob=0.1):
    probs = torch.nn.functional.softmax(logits.to(torch.float32), dim=-1).cpu()
    probs, indices = torch.topk(probs, k, dim=-1)

    # Pick the first token if all token probs are below `min_prob`
    pick_argmax = torch.all(probs < min_prob, dim=-1, keepdim=True)
    probs = probs.where((probs >= min_prob).logical_or(pick_argmax), 0.0)
    sampled = torch.multinomial(probs, num_samples=1)
    sampled.masked_fill_(pick_argmax, 0)

    return indices.gather(-1, sampled).to(torch.int32)


# FIXME, move to pytest
native_torch.set_fusion(4)
cache.compile_cache_dir = f"./.cache/{CACHE_KEY}/eager"
device = torch.device("rngd:0")

# Used by eager execution
compiler.set_default_compiler_config(
    compiler.Config(
        implicit_type_casting=True,
        implicit_type_casting_all_activation=True,
        allow_unlowered_operators=True,
        allow_external_operators_in_lir=True,
        tactic_hint=compiler.TacticHintConfig.Default,
        reshape_remover_ignores_slice_concat=False,
        max_num_unique_shapes=10,  # small size of unique shapes boosts compilation time
        tactic_sorting_policy=compiler.TacticSortingPolicy.Random,  # it boosts compilation time
    )
)

model = AutoModelForCausalLM.from_pretrained(
    MODEL,
    torch_dtype=torch.bfloat16,
    attn_implementation="eager",
    trust_remote_code=IS_REMOTE_CODE,
)
tokenizer = AutoTokenizer.from_pretrained(MODEL)
tokenizer.pad_token_id = 0
generation_config = GenerationConfig.from_pretrained(MODEL)

print("Tokenizing...")
if IS_INSTRUCT:
    system_prompt = "You are a friendly chatbot. Do not use emojis."
    prompt = [
        "Write a function that returns the greatest common divisor of two integers in Python. Do not use recursion.",
    ]
    chats = [
        [
            {"role": "system", "content": system_prompt},
            {"role": "user", "content": content},
        ]
        for content in prompt
    ]
    inputs = tokenizer.apply_chat_template(
        chats,
        return_tensors="pt",
        add_generation_prompt=True,
        padding=True,
        return_dict=True,
        tokenizer_kwargs={
            "padding_side": "left",
            "return_attention_mask": True,
        },
    )
else:
    prompt = ["Hello, my name is"]
    inputs = tokenizer(
        prompt,
        return_tensors="pt",
        padding=True,
        padding_side="left",
        return_attention_mask=True,
    )
num_batches = inputs.input_ids.shape[0]
length = inputs.input_ids.shape[1]
input_length = length
prefill_bucket = choose_bucket(length)

print("Copying to RNGD...")
model.eval()
model.to(device)

compile_model(model, cache_dir=f"./.cache/{CACHE_KEY}/b{num_batches}")

print("Warming up...")
warmup_start = time.monotonic()
warm_up(model, num_batches, prefill_bucket, MAX_SEQ_LEN, device)
warmup_end = time.monotonic()
print(f"Warmup took {warmup_end - warmup_start} seconds")

# Generate
print("Generating...")
input_ids = torch.zeros([num_batches, prefill_bucket], dtype=torch.int32)
input_ids[:, :length] = inputs.input_ids
is_running = torch.ones([num_batches], dtype=torch.bool)

# Prepare causal mask
mask_min = torch.finfo(torch.bfloat16).min
attention_mask = inputs.attention_mask.to(torch.bool)
prefill_causal_mask = (
    torch.ones([prefill_bucket, prefill_bucket], dtype=torch.bool)
    .triu(1)
    .expand(num_batches, 1, -1, -1)
    .contiguous()
)
prefill_causal_mask[:, 0, :, :length].bitwise_or_(
    attention_mask.bitwise_not()[:, None, :].expand(-1, prefill_bucket, -1)
)
prefill_causal_mask = prefill_causal_mask * torch.tensor(mask_min, dtype=torch.bfloat16)
decode_causal_mask = torch.full(
    [num_batches, 1, 1, MAX_SEQ_LEN], fill_value=mask_min, dtype=torch.bfloat16
)
decode_causal_mask[:, 0, 0, :length].masked_fill_(attention_mask, 0)

# Prefill
past_key_values = RngdStaticCache(
    config=model.config,
    max_batch_size=num_batches,
    max_cache_len=prefill_bucket,
    device=device,
    dtype=model.dtype,
)

args = {
    "input_ids": input_ids.to(device),
    "attention_mask": prefill_causal_mask.to(device),
    "use_cache": True,
    "cache_position": torch.arange(prefill_bucket, dtype=torch.int32).to(device),
    "past_key_values": past_key_values,
    "return_dict": True,
}

output_ids = torch.zeros([num_batches, MAX_SEQ_LEN], dtype=torch.int32)
output_ids[:, :length] = inputs.input_ids
prefill = True
next_input_ids = torch.zeros([num_batches, 1], dtype=torch.int32)

streamer = (
    TextStreamer(tokenizer, skip_special_tokens=True) if num_batches == 1 else None
)

generation_start = time.monotonic()
prefill_end = generation_start
num_generated_tokens = 0
while length < MAX_SEQ_LEN and torch.any(is_running):
    inference_result = model(**args)
    next_idx = length - 1 if prefill else -1
    logits = inference_result.logits[:, next_idx, :]
    next_input_ids = sample_topk(logits, 5).where(
        is_running.unsqueeze(1), tokenizer.pad_token_id
    )
    if prefill:
        prefill_end = time.monotonic()
        if streamer:
            if IS_INSTRUCT:
                print(f"<Sys> {system_prompt}")
                print(f"<Usr> {prompt[0]}")
            else:
                streamer.put(output_ids[0, :length])
        args["attention_mask"] = torch.empty(
            [num_batches, 1, 1, MAX_SEQ_LEN], dtype=torch.bfloat16, device=device
        )
        past_key_values.resize(MAX_SEQ_LEN)

    if streamer:
        streamer.put(next_input_ids[0])

    output_ids[:, length] = next_input_ids[:, 0]
    decode_causal_mask[:, 0, 0, length].masked_fill_(is_running, 0)
    args["input_ids"] = next_input_ids.to(device)
    args["attention_mask"].copy_(decode_causal_mask)
    args["cache_position"] = torch.tensor([length], dtype=torch.int32).to(device)

    prefill = False
    length += 1

    num_generated_tokens += is_running.sum()
    cont = next_input_ids.squeeze(1) != tokenizer.eos_token_id
    is_running.bitwise_and_(cont)
generation_end = time.monotonic()

if streamer:
    streamer.end()
    print("==========")
else:
    decode_from = input_length if IS_INSTRUCT else 0
    decoded = tokenizer.batch_decode(
        output_ids[:, decode_from:],
        skip_special_tokens=True,
        clean_up_tokenization_spaces=False,
    )
    for output in decoded:
        print(output)
        print("==========")

if IS_INSTRUCT:
    print(f"System prompt: {system_prompt}")
print(f"Input prompt ({num_batches}-batch):")
for p in prompt:
    print(f"- {p}")
print(f"Total number of generated tokens: {num_generated_tokens}")
print(f"Prefill took {prefill_end - generation_start} seconds")
decode_time = generation_end - prefill_end
print(
    f"Decode took {decode_time} seconds ({(num_generated_tokens - num_batches) / decode_time} TPS)"
)
