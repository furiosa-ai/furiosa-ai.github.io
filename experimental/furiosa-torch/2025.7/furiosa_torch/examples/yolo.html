import os
import pathlib

import cv2
import numpy as np
import torch
from furiosa import native_torch
from furiosa.native_torch import compiler
from furiosa.torch import backend, cache
from furiosa.torch.profiler import RNGDProfiler
from ultralytics import YOLO
from ultralytics.models.yolo.detect.predict import DetectionPredictor
import ultralytics.nn.modules.block


def patch_ultralytics_modules():
    """
    Patch Ultralytics modules to be TorchDynamo-friendly.

    Specifically, expands generator pattern like `m(y[-1]) for m in lst` into for-loop.
    TorchDynamo thinks `y[-1]` has fixed value, but actually it is not when the generator is used
    with `y.extend`.
    """

    def c2f_forward(self, x):
        y = list(self.cv1(x).chunk(2, 1))
        for m in self.m:
            y.append(m(y[-1]))
        return self.cv2(torch.cat(y, 1))

    def c2f_forward_split(self, x):
        y = self.cv1(x).split((self.c, self.c), 1)
        y = [y[0], y[1]]
        for m in self.m:
            y.append(m(y[-1]))
        return self.cv2(torch.cat(y, 1))

    def sppf_forward(self, x):
        y = [self.cv1(x)]
        for _ in range(3):
            y.append(self.m(y[-1]))
        return self.cv2(torch.cat(y, 1))

    ultralytics.nn.modules.block.C2f.forward = c2f_forward
    ultralytics.nn.modules.block.C2f.forward_split = c2f_forward_split
    ultralytics.nn.modules.block.SPPF.forward = sppf_forward


# Turn on inline inbuilt nn modules to reduce pass all weights as parameters
torch._dynamo.config.inline_inbuilt_nn_modules = False
patch_ultralytics_modules()

MODEL = "yolov8n.pt"
CACHE_KEY = "yolov8n"

# yolo has a bug that missing dtype field,
# https://github.com/ultralytics/ultralytics/pull/20096

# Used by eager execution

        implicit_type_casting_all_activation=True,

        reshape_remover_ignores_slice_concat=True,
        max_num_unique_shapes=100,

)

    implicit_type_casting_all_activation=True,
    allow_unlowered_operators=True,
    allow_external_operators_in_lir=True,

class RngdDetectionPredictor(DetectionPredictor):
    """DetectionPredictor with custom preprocess and postprocess methods."""

    def preprocess(self, im):
        """Copy of BasePredictor.preprocess with slight changes."""

        not_tensor = not isinstance(im, torch.Tensor)
        if not_tensor:
            # Force letterboxing
            im = np.stack(self.pre_transform([*im, np.zeros((*self.imgsz, 3))])[:-1])
            im = im[..., ::-1].transpose((0, 3, 1, 2))
            im = np.ascontiguousarray(im)
            im = torch.from_numpy(im)

        # Send to device after casting and division.
        # FIXME: maybe boost compilation speed of div.Scalar?
        im = im.half() if self.model.fp16 else im.float()
        if not_tensor:
            im /= 255.0
        return im.to(self.device)

    def postprocess(self, preds, img, orig_img, **kwargs):
        def to_cpu(x):
            if isinstance(x, torch.Tensor):
                return x.cpu()
            elif isinstance(x, list):
                return [to_cpu(x) for x in x]
            elif isinstance(x, tuple):
                return (to_cpu(x) for x in x)
            else:
                return x

        return super().postprocess(to_cpu(preds), img, orig_img, **kwargs)


with RNGDProfiler(
    schedule=torch.profiler.schedule(
        wait=wait,
        warmup=warmup,
        active=active,
    )
) as profiler:
    yolo = YOLO(MODEL)
    yolo.eval()

    # Warm up on CPU first to avoid graph breaks and recompilations
    results = yolo.predict(
        img,
        device=torch.device("cpu"),
        predictor=RngdDetectionPredictor,
    )

    yolo = yolo.to(device=device)
    yolo.model.compile(
        backend=backend.backend_with(
            compiler_config=compiler_config,
            # Set cache directory to utilize already compiled ones.
            cache_dir=f"./.cache/{CACHE_KEY}",
        ),
        dynamic=False,
    )

    # Wait
    profiler.step()

    # Warm up
    results = yolo.predict(img, device=device, predictor=RngdDetectionPredictor)
    profiler.step()

    # inference
    for idx in range(active):
        print(f"iter {idx}")
        results = yolo.predict(img, device=device, predictor=RngdDetectionPredictor)
        profiler.step()

profiler.export_chrome_trace(f"profile_{CACHE_KEY}.json")
