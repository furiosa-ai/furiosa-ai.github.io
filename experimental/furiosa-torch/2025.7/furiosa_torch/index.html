furiosa-torch in general

furiosa-torch in general#

  1"""
  2furiosa.torch consists of 5 parts:
  31. native supports torch by providing a torch custom backend for RNGD devices.
  42. supports torch.compile to utilize FuriosaAI compiler, which optimizes based on whole graph compilation.
  53. provides hal which is a programming interface of hardware execution models.
  64. provides compliation interfaces and execution interfaces for the RNGD runnable.
  75. provides lla which is a low-level programming interface for each hardware components.
  8"""
  9
 10"""
 111. torch
 12Each torch operation against RNGD tensor, will be compiled to RNGD runnable, and run on RNGD.
 13"""
 14import furiosa.torch as _
 15import torch
 16
 17device = torch.device("rngd:0")
 18
 19a = torch.rand([32, 2, 256], dtype=torch.bfloat16)
 20b = torch.rand([32, 2, 256], dtype=torch.bfloat16)
 21c = a.to(device) + b.to(device)  # run on RNGD
 22assert torch.all(c.cpu() == a + b)
 23
 24
 25"""
 262. torch.compile
 27torch.compile will traces target model into a fx.GraphModule, 
 28which FuriosaAI compiler takes as an input format of compilation.
 29"""
 30import furiosa.torch.backend
 31
 32
 33def model(a, b):
 34    return (a - b) * (a + b)
 35
 36
 37a = torch.rand([32, 2, 256], dtype=torch.bfloat16)
 38b = torch.rand([32, 2, 256], dtype=torch.bfloat16)
 39model = torch.compile(model, backend=furiosa.torch.backend.backend)
 40z = model(a.to(device), b.to(device))
 41assert torch.allclose(z.cpu(), (a - b) * (a + b), atol=0.1, rtol=0.1)
 42
 43
 44"""
 453. hal
 46Descriptive hardware manipuliation API, implements software-out-of-order 
 47along hal programs, and parallel joints of pe programs
 48"""
 49from furiosa import native_torch
 50from furiosa.native_torch import hal, ir
 51
 52rngd = native_torch.get_device(device.index)
 53tensor_in_cpu_0 = torch.rand([32, 2, 256], dtype=torch.bfloat16)
 54tensor_in_dram_0 = torch.empty([32, 2, 256], dtype=torch.bfloat16, device=device)
 55tensor_in_dram_1 = torch.empty([32, 2, 256], dtype=torch.bfloat16, device=device)
 56tensor_in_cpu_1 = torch.empty([32, 2, 256], dtype=torch.bfloat16)
 57
 58hal_pe_program = hal.pert.seq(
 59    [
 60        hal.pert.copy_local(
 61            tensor_in_dram_0,
 62            tensor_in_dram_1,
 63            ir.Sequencer([(0, 1, 32, 512), (1, 1, 2, 256), (2, 1, 256, 1)]),
 64            ir.Sequencer([(0, 1, 32, 512), (1, 1, 2, 256), (2, 1, 256, 1)]),
 65        )
 66    ]
 67)
 68
 69hal_program = hal.seq(
 70    [
 71        hal.copy(tensor_in_cpu_0, tensor_in_dram_0),
 72        hal.execute(hal_pe_program),
 73        hal.copy(tensor_in_dram_1, tensor_in_cpu_1),
 74    ]
 75)
 76
 77rngd.execute_blocking(hal_program)
 78assert torch.all(tensor_in_cpu_0 == tensor_in_cpu_1)
 79

 81"""
 824. compile
 83FuriosaAI compiler takes GraphModule as an input format of compilation, and creates EDF
 84which is RNGD runnable with metadata to run properly.
 85hal program, and hal pe program describes how to run the runnable on RNGD.
 86"""
 87import torch.fx
 88from furiosa.torch import compiler, executor
 89
 90device = torch.device("rngd:0")
 91
 92# Build (a + b) + b
 93g = torch.fx.Graph()
 94i0 = g.placeholder("i0")
 95i1 = g.placeholder("i1")
 96o0 = g.call_function(torch.ops.aten.add, (i0, i1))
 97o1 = g.call_function(torch.ops.aten.add, (o0, i1))
 98g.output(o1)
 99gm = torch.fx.GraphModule(torch.nn.Module(), g)
100
101a = torch.rand([32, 2, 256], dtype=torch.bfloat16)
102b = torch.rand([32, 2, 256], dtype=torch.bfloat16)
103inputs = [a.to(device), b.to(device)]
104edf = compiler.compile_gm(gm, inputs)
105execute = executor.load_edf(edf, device)
106# It returns outputs & profiles
107outputs, _ = execute(inputs)
108
109assert torch.all(outputs[0].cpu() == (a + b) + b)
110
111"""
1125. GraphModule eager execution via fx.Interpreter
113"""
114import furiosa.torch as _
115import torch
116import torch.fx
117
118# Build (a + b) + b
119g = torch.fx.Graph()
120i0 = g.placeholder("i0")
121i1 = g.placeholder("i1")
122o0 = g.call_function(torch.ops.aten.add, (i0, i1))
123o1 = g.call_function(torch.ops.aten.add, (o0, i1))
124g.output(o1)
125gm = torch.fx.GraphModule(torch.nn.Module(), g)
126interpreter = torch.fx.Interpreter(gm)
127
128a = torch.rand([32, 2, 256], dtype=torch.bfloat16)
129b = torch.rand([32, 2, 256], dtype=torch.bfloat16)
130output = interpreter.run(a.to(device), b.to(device))
131
132assert torch.all(output.cpu() == (a + b) + b)
133
134"""
1356. lla
136Low-level programming interface for each hardware components.
137Each component has 'output_shape', and 'output_dtype', hardware component takes previous component's 
138'output_shape', and 'output_dtype' as 'input_shape', and 'input_dtype'.
139e.g., 
140fetch_engine = lla.FetchEngine(input_shape=ir.LoweredShape(..), ...)
141tensor_engine = lla.TensorEngine(input_shape=fetch_unit.output_shape, ...)
142global_adder_tree_engine = lla.GlobalAdderTreeEngine(input_shape=tensor_engine.output_shape, ...)
143"""
144from furiosa.native_torch import ir, lla
145from furiosa.torch.lla import (
146    compile_and_load_tu_exec,
147    execute_sfr,
148    tensor_to_sram,
149    tensor_to_dram,
150)
151
152device = torch.device("rngd:0")
153
154input_cpu_tensor = torch.rand([32, 2, 256], dtype=torch.bfloat16)
155input_dram_tensor = input_cpu_tensor.to(device=device)
156input_sram_tensor = tensor_to_sram(input_dram_tensor)
157
158tu_exec = lla.TuExec(
159    fetch_engine=lla.FetchEngine(
160        input_shape=ir.LoweredShape(partitioning=[32, 2], in_slice=[256]),
161        input_dtype=torch.bfloat16,
162    ),
163    tensor_engine=None,
164    global_adder_tree_engine=None,
165    vector_engine=None,
166    cast_compaction_engine=None,
167    transpose_engine=None,
168    commit_engine=None,
169)
170# creates EDF from tu_exec, and run on RNGD
171sfr_addr, sfr_size_in_bytes, subunit_bitmap, _ = compile_and_load_tu_exec(
172    tu_exec, device
173)
174output_sram_tensor = execute_sfr(
175    sfr_addr,
176    sfr_size_in_bytes,
177    subunit_bitmap,
178    input_sram_tensor,
179    (tu_exec.output_shape.shape(), tu_exec.output_dtype),
180)
181output_cpu_tensor = tensor_to_dram(output_sram_tensor).cpu().squeeze()
182assert torch.all(input_cpu_tensor == output_cpu_tensor)
