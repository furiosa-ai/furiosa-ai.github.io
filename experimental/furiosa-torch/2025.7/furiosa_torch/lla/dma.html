DMA (DRAM <> SRAM)

DMA (DRAM <> SRAM)#

"""
Before executing any of RNGD low-level API, all tensor must be loaded in SRAM.
This example is how to load tensors to SRAM properly. RNGD has 64 slices per PE,
SRAM tensor is aliged to slices. When you use fused PE (e.g., 2PE fusion), SRAM must be
partitioned to 128.
"""

import asyncio

import torch
from furiosa import native_torch
from furiosa.native_torch import hal, ir
from furiosa.torch.lla import empty_sram, tensor_to_sram

dram_device = torch.device("rngd:0")

dram_tensor = torch.rand([128, 256]).to(dram_device)
# To load tensor to SRAM, it must be partitioned into num slices (in this case, we use 2 fused PE, so 64 * 2 slices)
# Each slice has [2, 256] tensor.
# NOTE: we will provide flexible sequencing between DRAM <> SRAM in the future.
sram_tensor = tensor_to_sram(dram_tensor.reshape([64, 2, 256]))

# You can split into slices by reshape the slices axis, e.g., [64] -> [4, 16]
dram_tensor = dram_tensor.reshape([4, 16, 2, 256])
# It indicates first 32 slices will have dram_tensor[0][0:32],
# and the next 32 slices will have dram_tensor[1][0:32], and so on.
sram_tensor = tensor_to_sram(dram_tensor)

# When the shape is not aligned to 64 , it will arise an exception.
try:
    sram_tensor = tensor_to_sram(dram_tensor.reshape([256, 256]))
except Exception as e:
    print(e)

# Or you can copy DRAM <> SRAM use raw API in HAL (copy_local)
dram_tensor = torch.rand([64, 16, 256]).to(dram_device)
sram_tensor = empty_sram([64, 16, 256], device=dram_device)
rngd = native_torch.get_device(dram_device.index)
asyncio.run(
    rngd.execute(
        hal.execute(
            hal.pert.copy_local(
                dram_tensor,
                sram_tensor,
                ir.Sequencer.from_tensor(dram_tensor, dram_partition_by_pe=True),
                ir.Sequencer.from_tensor(sram_tensor, dram_partition_by_pe=True),
            )
        )
    )
)
